### 第 4 步：更新小程序页面

**文件：** `miniprogram/pages/index/index.js`

```javascript
import { getMerchants } from '../../api/request'

Page({
  data: {
    merchants: [],
    loading: true
  },

  onLoad() {
    this.loadMerchants()
  },

  async loadMerchants() {
    try {
      this.setData({ loading: true })
      const res = await getMerchants()
      
      if (res.code === '0000') {
        this.setData({
          merchants: res.data.merchants,
          loading: false
        })
      } else {
        wx.showToast({
          title: '加载失败',
          icon: 'error'
        })
      }
    } catch (error) {
      console.error('加载商家失败:', error)
      wx.showToast({
        title: '网络错误',
        icon: 'error'
      })
    }
  }
})
```

### 第 5 步：更新 Streamlit 大屏

**文件：** `cloud_server/god_dashboard.py`

```python
import streamlit as st
import requests
import pandas as pd

# 设置页面配置
st.set_page_config(page_title="Project Claw", layout="wide")

# 获取统计数据
@st.cache_data
def get_stats():
    try:
        response = requests.get('http://localhost:8765/api/v1/statistics/dashboard')
        if response.status_code == 200:
            return response.json()['data']
    except:
        pass
    return {}

# 侧边栏菜单
page = st.sidebar.radio(
    "选择页面",
    ["📊 仪表板", "💬 对话管理", "🏪 商家管理", "📈 数据分析"]
)

# 获取数据
stats = get_stats()

if page == "📊 仪表板":
    st.title("融资路演大屏")
    
    col1, col2, col3, col4 = st.columns(4)
    with col1:
        st.metric("在线商家", stats.get('online_merchants', 0))
    with col2:
        st.metric("活跃对话", stats.get('active_dialogues', 0))
    with col3:
        st.metric("总销售额", f"¥{stats.get('total_sales', 0):,}")
    with col4:
        st.metric("平均评分", f"{stats.get('avg_rating', 0):.1f}⭐")

elif page == "💬 对话管理":
    st.title("对话管理")
    st.write(f"总对话数: {stats.get('total_dialogues', 0)}")
    st.write(f"已完成: {stats.get('completed_dialogues', 0)}")

elif page == "🏪 商家管理":
    st.title("商家管理")
    st.write(f"在线商家: {stats.get('online_merchants', 0)}")
    
    # 按分类显示
    categories = stats.get('merchants_by_category', {})
    for category, merchants in categories.items():
        st.subheader(f"{category} ({len(merchants)})")

elif page == "📈 数据分析":
    st.title("数据分析")
    st.write("数据分析页面")
```
