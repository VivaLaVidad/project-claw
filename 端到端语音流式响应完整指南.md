### 第 2 步：小程序端实现

```javascript
// miniprogram/utils/audio_stream.js

class AudioStreamClient {
    constructor(serverUrl) {
        this.serverUrl = serverUrl
        this.ws = null
        this.audioContext = null
        this.mediaRecorder = null
        this.sessionId = null
    }
    
    async init() {
        // 初始化 WebAudio API
        this.audioContext = new (window.AudioContext || window.webkitAudioContext)()
        
        // 请求麦克风权限
        const stream = await navigator.mediaDevices.getUserMedia({ audio: true })
        
        // 创建 MediaRecorder
        this.mediaRecorder = new MediaRecorder(stream)
        
        // 处理录制数据
        this.mediaRecorder.ondataavailable = (event) => {
            this.sendAudioChunk(event.data)
        }
    }
    
    connect(sessionId) {
        this.sessionId = sessionId
        const wsUrl = `${this.serverUrl}/ws/audio/${sessionId}`
        
        this.ws = new WebSocket(wsUrl)
        this.ws.binaryType = 'arraybuffer'
        
        this.ws.onopen = () => {
            console.log('WebSocket 已连接')
            this.startRecording()
        }
        
        this.ws.onmessage = (event) => {
            if (event.data instanceof ArrayBuffer) {
                // 接收音频数据
                this.playAudio(event.data)
            } else {
                // 接收元数据
                const metadata = JSON.parse(event.data)
                console.log('音频元数据:', metadata)
            }
        }
        
        this.ws.onerror = (error) => {
            console.error('WebSocket 错误:', error)
        }
        
        this.ws.onclose = () => {
            console.log('WebSocket 已断开')
            this.stopRecording()
        }
    }
    
    startRecording() {
        this.mediaRecorder.start(100) // 每 100ms 发送一次
    }
    
    stopRecording() {
        if (this.mediaRecorder) {
            this.mediaRecorder.stop()
        }
    }
    
    sendAudioChunk(blob) {
        if (this.ws && this.ws.readyState === WebSocket.OPEN) {
            // 转换为 ArrayBuffer
            const reader = new FileReader()
            reader.onload = () => {
                this.ws.send(reader.result)
            }
            reader.readAsArrayBuffer(blob)
        }
    }
    
    playAudio(audioData) {
        // 解码音频
        this.audioContext.decodeAudioData(audioData, (buffer) => {
            // 创建音频源
            const source = this.audioContext.createBufferSource()
            source.buffer = buffer
            source.connect(this.audioContext.destination)
            source.start(0)
        })
    }
    
    disconnect() {
        if (this.ws) {
            this.ws.close()
        }
        this.stopRecording()
    }
}

export default AudioStreamClient
```

### 第 3 步：小程序页面集成

```javascript
// miniprogram/pages/voice_chat/voice_chat.js

import AudioStreamClient from '../../utils/audio_stream'

Page({
    data: {
        isConnected: false,
        isRecording: false,
        sessionId: null
    },
    
    onLoad() {
        const app = getApp()
        this.audioClient = new AudioStreamClient(app.globalData.serverBase)
    },
    
    async onStartVoiceChat() {
        try {
            // 初始化音频
            await this.audioClient.init()
            
            // 生成会话 ID
            const sessionId = `voice_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`
            
            // 连接 WebSocket
            this.audioClient.connect(sessionId)
            
            this.setData({
                isConnected: true,
                isRecording: true,
                sessionId: sessionId
            })
            
            wx.showToast({
                title: '语音对讲已启动',
                icon: 'success'
            })
        } catch (error) {
            console.error('启动语音对讲失败:', error)
            wx.showToast({
                title: '启动失败',
                icon: 'error'
            })
        }
    },
    
    onStopVoiceChat() {
        this.audioClient.disconnect()
        
        this.setData({
            isConnected: false,
            isRecording: false
        })
        
        wx.showToast({
            title: '语音对讲已结束',
            icon: 'success'
        })
    }
})
```
