## 📋 核心功能

### 1. 断点挂起机制

```python
# edge_box/agent_workflow.py

from langgraph.graph import StateGraph
from langgraph.checkpoint.memory import MemorySaver

class AgentWorkflow:
    def __init__(self):
        self.graph = StateGraph(AgentState)
        self.checkpointer = MemorySaver()
    
    def build_graph(self):
        # 添加节点
        self.graph.add_node("critic", self.critic_node)
        self.graph.add_node("execute", self.execute_node)
        
        # 添加条件边
        self.graph.add_conditional_edges(
            "critic",
            self.should_suspend,
            {
                "suspend": "suspend",
                "execute": "execute"
            }
        )
        
        # 编译图
        self.compiled = self.graph.compile(
            checkpointer=self.checkpointer,
            interrupt_before=["execute"]  # 在执行前中断
        )
    
    async def critic_node(self, state: AgentState):
        """评估节点"""
        confidence = state.get("confidence", 1.0)
        
        if confidence < 0.7:
            # 触发挂起
            await self.suspend_task(state)
            return {"suspended": True}
        
        return {"suspended": False}
    
    async def suspend_task(self, state: AgentState):
        """挂起任务"""
        # 序列化状态
        state_data = json.dumps(state)
        
        # 推入 Redis
        await redis_client.lpush(
            "hitl_pending_queue",
            state_data
        )
        
        logger.info(f"✓ 任务已挂起: {state['task_id']}")
```

### 2. 轻量级遥测

```python
# edge_box/telemetry_client.py

import paho.mqtt.client as mqtt
import json
import time

class TelemetryClient:
    def __init__(self, box_id: str):
        self.box_id = box_id
        self.client = mqtt.Client(mqtt.CallbackAPIVersion.VERSION1)
        self.client.connect("localhost", 1883, 60)
        self.client.loop_start()
    
    async def send_telemetry(self):
        """每 10 秒发送一次遥测"""
        while True:
            telemetry = {
                "box_id": self.box_id,
                "status": self.get_status(),
                "daily_orders": self.get_daily_orders(),
                "timestamp": time.time()
            }
            
            self.client.publish(
                "edge_box/telemetry/status",
                json.dumps(telemetry)
            )
            
            await asyncio.sleep(10)
    
    def get_status(self) -> str:
        """获取盒子状态"""
        # idle, busy, error, suspended
        return "idle"
    
    def get_daily_orders(self) -> int:
        """获取今日订单数"""
        return 0
```

```python
# cloud_server/api_server_pro.py

from fastapi import FastAPI, APIRouter
from cloud_server.fleet_manager import FleetManager
from cloud_server.hitl_triage import HITLTriage

app = FastAPI()
fleet_manager = FleetManager()
hitl_triage = HITLTriage()

@app.get("/fleet/pending-tasks")
async def get_pending_tasks(limit: int = 100):
    """获取所有待审批任务"""
    tasks = await fleet_manager.get_pending_tasks(limit)
    return {"success": True, "data": tasks}

@app.post("/fleet/approve-task")
async def approve_task(decision: Dict[str, Any]):
    """批准任务"""
    from cloud_server.fleet_manager import ApprovalDecision
    
    approval = ApprovalDecision(**decision)
    success = await fleet_manager.approve_task(approval)
    
    if success:
        return {"success": True, "message": "任务已批准"}
    else:
        raise HTTPException(status_code=500, detail="批准失败")

@app.get("/fleet/status")
async def get_fleet_status():
    """获取舰队状态"""
    status = await fleet_manager.get_fleet_status()
    return {"success": True, "data": status}

@app.get("/fleet/triage/stats")
async def get_triage_stats():
    """获取分流统计"""
    stats = await hitl_triage.get_queue_stats()
    return {"success": True, "data": stats}
```

## 📊 工作流程

### 完整的任务挂起和恢复流程

```
1. 边缘设备执行任务
   ↓
2. CriticNode 评估置信度
   ↓
3. 如果置信度 < 0.7
   ├─ 序列化状态
   ├─ 推入 Redis hitl_pending_queue
   ├─ 发送 MQTT 遥测
   └─ 暂停 LangGraph 状态机
   ↓
4. 云端指挥中心接收
   ├─ FleetManager 注册挂起任务
   ├─ HITLTriage 评估优先级
   └─ 路由到相应队列
   ↓
5. 人工审批
   ├─ 查看待审批任务
   ├─ 做出决策 (Accept/Override/Reject)
   └─ 提交批准
   ↓
6. 云端发布决策
   ├─ 通过 Redis Pub/Sub 发布
   └─ 边缘设备订阅并接收
   ↓
7. 边缘设备恢复执行
   ├─ 从 Redis 恢复状态
   ├─ 应用人工决策
   └─ 继续执行任务
   ↓
8. 任务完成
```
