  ↓

## 🚀 使用示例

### 在页面中使用

```javascript
// pages/index/index.js
const app = getApp()

Page({
  onLoad() {
    // 获取核心模块
    this.eventBus = app.getEventBus()
    this.stateManager = app.getStateManager()
    this.tradeEngine = app.getTradeEngine()
    this.performanceMonitor = app.getPerformanceMonitor()

    // 订阅状态变化
    this.unsubscribeTrade = this.stateManager.subscribe('currentTrade', (trade) => {
      console.log('询价已更新:', trade)
      this.setData({ currentTrade: trade })
    })

    // 监听事件
    this.eventBus.on('trade:completed', (trade) => {
      console.log('询价已完成:', trade)
      wx.showToast({ title: '询价完成', icon: 'success' })
    })

    this.eventBus.on('network:offline', () => {
      wx.showToast({ title: '网络已断开', icon: 'none' })
    })
  },

  async submitTrade() {
    // 创建询价请求
    const request = this.tradeEngine.createTradeRequest({
      client_id: app.globalData.clientId,
      item_name: this.data.form.item_name,
      demand_text: this.data.form.demand_text,
      max_price: Number(this.data.form.max_price),
      quantity: Number(this.data.form.quantity),
      timeout_sec: Number(this.data.form.timeout_sec)
    })

    // 更新状态
    this.tradeEngine.updateTradeStatus('pending')

    // 发送请求
    try {
      const bundle = await requestTrade(request)
      
      // 处理报价
      this.tradeEngine.handleOffers(bundle.offers || [])
      
      // 完成询价
      this.tradeEngine.completeTrade(bundle)
    } catch (error) {
      console.error('询价失败:', error)
      this.tradeEngine.updateTradeStatus('failed', { error: error.message })
    }
  },

  onUnload() {
    // 取消订阅
    this.unsubscribeTrade?.()
    
    // 移除事件监听
    this.eventBus.clear('trade:completed')
    this.eventBus.clear('network:offline')
  }
})
```
