```javascript
// 获取服务器地址
function getServerBase() {
  return 'http://localhost:8765'
}

// 通用请求函数
export function request(options) {
  return new Promise((resolve, reject) => {
    wx.request({
      url: getServerBase() + options.url,
      method: options.method || 'GET',
      data: options.data || {},
      header: {
        'Content-Type': 'application/json'
      },
      success: (res) => {
        if (res.statusCode === 200) {
          resolve(res.data)
        } else {
          console.error('API错误:', res.statusCode, res.data)
          reject(res.data)
        }
      },
      fail: (err) => {
        console.error('网络错误:', err)
        reject(err)
      }
    })
  })
}

// 获取所有商家
export function getMerchants() {
  return request({
    url: '/api/v1/merchants',
    method: 'GET'
  })
}

// 获取商家详情
export function getMerchant(merchantId) {
  return request({
    url: `/api/v1/merchants/${merchantId}`,
    method: 'GET'
  })
}

// 按分类获取商家
export function getMerchantsByCategory(category) {
  return request({
    url: `/api/v1/merchants/category/${category}`,
    method: 'GET'
  })
}

// 创建对话
export function createDialogue(data) {
  return request({
    url: '/api/v1/dialogues',
    method: 'POST',
    data
  })
}

// 获取对话
export function getDialogue(sessionId) {
  return request({
    url: `/api/v1/dialogues/${sessionId}`,
    method: 'GET'
  })
}

// 添加消息
export function addMessage(sessionId, data) {
  return request({
    url: `/api/v1/dialogues/${sessionId}/messages`,
    method: 'POST',
    data
  })
}

// 获取仪表板统计
export function getDashboardStats() {
  return request({
    url: '/api/v1/statistics/dashboard',
    method: 'GET'
  })
}

// 获取健康状态
export function getHealth() {
  return request({
    url: '/api/v1/health',
    method: 'GET'
  })
}
```

### 第 2 步：修复小程序首页

**文件：** `miniprogram/pages/index/index.js`

```javascript
import { getMerchants, getHealth } from '../../api/request'

Page({
  data: {
    merchants: [],
    loading: true,
    error: null,
    serverConnected: false
  },

  onLoad() {
    this.checkServer()
    this.loadMerchants()
  },

  async checkServer() {
    try {
      const res = await getHealth()
      if (res.code === '0000') {
        this.setData({ serverConnected: true })
        console.log('✓ 服务器连接成功')
      }
    } catch (error) {
      console.error('✗ 服务器连接失败:', error)
      this.setData({ serverConnected: false })
    }
  },

  async loadMerchants() {
    try {
      this.setData({ loading: true, error: null })
      
      const res = await getMerchants()
      
      if (res.code === '0000') {
        const merchants = res.data.merchants || []
        this.setData({
          merchants: merchants,
          loading: false
        })
        console.log(`✓ 加载 ${merchants.length} 个商家`)
      } else {
        throw new Error(res.message || '加载失败')
      }
    } catch (error) {
      console.error('加载商家失败:', error)
      this.setData({
        error: error.message || '加载失败',
        loading: false,
        merchants: []
      })
      wx.showToast({
        title: '加载失败',
        icon: 'error'
      })
    }
  },

  onMerchantTap(e) {
    const merchantId = e.currentTarget.dataset.id
    wx.navigateTo({
      url: `/pages/merchant/merchant?id=${merchantId}`
    })
  },

  onRefresh() {
    this.loadMerchants()
  }
})
```

### 第 3 步：修复后端 API 路由

**文件：** `cloud_server/api_server_pro.py`

```python
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware

from .api_routes_fix import router as api_v1_router
from .industrial_fix import (
    get_merchants_api,
    get_merchant_api,
    get_dashboard_stats_api,
    get_dialogues_api
)

app = FastAPI(
    title="Project Claw API",
    description="Project Claw 后端 API 服务",
    version="1.0.0"
)

# CORS 配置
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# 注册 API 路由
app.include_router(api_v1_router)

# 健康检查
@app.get("/api/v1/health")
async def health_check():
    """健康检查"""
    return {
        "code": "0000",
        "message": "success",
        "data": {"status": "healthy"}
    }

# 获取所有商家
@app.get("/api/v1/merchants")
async def get_merchants():
    """获取所有在线商家"""
    return await get_merchants_api()

# 获取单个商家
@app.get("/api/v1/merchants/{merchant_id}")
async def get_merchant(merchant_id: str):
    """获取单个商家"""
    return await get_merchant_api(merchant_id)

# 获取仪表板统计
@app.get("/api/v1/statistics/dashboard")
async def get_stats():
    """获取仪表板统计"""
    return await get_dashboard_stats_api()

# 获取所有对话
@app.get("/api/v1/dialogues")
async def get_dialogues():
    """获取所有对话"""
    return await get_dialogues_api()

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8765)
```

### 第 4 步：修复 Streamlit 大屏

**文件：** `cloud_server/god_dashboard.py`

```python
import streamlit as st
import requests
import pandas as pd
from datetime import datetime

# 设置页面配置
st.set_page_config(
    page_title="Project Claw - 融资路演大屏",
    page_icon="🦞",
    layout="wide",
    initial_sidebar_state="expanded"
)

# 自定义样式
st.markdown("""
<style>
    .main {
        padding: 2rem;
    }
    .metric-card {
        background: linear-gradient(135deg, #667eea 0%, #764ba2 100%);
        padding: 1.5rem;
        border-radius: 10px;
        color: white;
        text-align: center;
    }
</style>
""", unsafe_allow_html=True)

# 获取统计数据
@st.cache_data(ttl=10)
def get_stats():
    try:
        response = requests.get('http://localhost:8765/api/v1/statistics/dashboard', timeout=5)
        if response.status_code == 200:
            return response.json().get('data', {})
    except Exception as e:
        st.error(f"获取数据失败: {e}")
    return {}

# 侧边栏菜单
st.sidebar.title("🦞 Project Claw")
page = st.sidebar.radio(
    "选择页面",
    ["📊 仪表板", "💬 对话管理", "🏪 商家管理", "📈 数据分析", "⚙️ 系统设置"]
)

# 获取数据
stats = get_stats()

# 页面 1：仪表板
if page == "📊 仪表板":
    st.title("融资路演大屏")
    
    # 关键指标
    col1, col2, col3, col4 = st.columns(4)
    
    with col1:
        st.metric(
            "在线商家",
            stats.get('online_merchants', 0),
            delta=f"总计 {stats.get('total_merchants', 0)}"
        )
    
    with col2:
        st.metric(
            "活跃对话",
            stats.get('active_dialogues', 0),
            delta=f"已完成 {stats.get('completed_dialogues', 0)}"
        )
    
    with col3:
        st.metric(
            "总销售额",
            f"¥{stats.get('total_sales', 0):,}",
            delta="今日数据"
        )
    
    with col4:
        st.metric(
            "平均评分",
            f"{stats.get('avg_rating', 0):.1f}⭐",
            delta="商家评分"
        )
    
    st.divider()
    
    # 商家分布
    st.subheader("商家分布")
    categories = stats.get('merchants_by_category', {})
    if categories:
        category_data = {cat: len(merchants) for cat, merchants in categories.items()}
        st.bar_chart(category_data)
    else:
        st.info("暂无商家数据")

# 页面 2：对话管理
elif page == "💬 对话管理":
    st.title("对话管理")
    
    col1, col2 = st.columns(2)
    with col1:
        st.metric("总对话数", stats.get('total_dialogues', 0))
    with col2:
        st.metric("进行中", stats.get('active_dialogues', 0))
    
    st.divider()
    st.info("对话管理功能")

# 页面 3：商家管理
elif page == "🏪 商家管理":
    st.title("商家管理")
    
    st.metric("在线商家", stats.get('online_merchants', 0))
    
    st.divider()
    
    # 按分类显示
    categories = stats.get('merchants_by_category', {})
    if categories:
        for category, merchants in categories.items():
            st.subheader(f"{category.upper()} ({len(merchants)})")
            for merchant in merchants:
                col1, col2, col3 = st.columns(3)
                with col1:
                    st.write(f"**{merchant.get('shop_name')}**")
                with col2:
                    st.write(f"⭐ {merchant.get('rating')}")
                with col3:
                    st.write(f"📊 {merchant.get('sales')} 笔")
    else:
        st.info("暂无商家数据")

# 页面 4：数据分析
elif page == "📈 数据分析":
    st.title("数据分析")
    
    st.write("### 关键指标")
    st.json(stats)

# 页面 5：系统设置
elif page == "⚙️ 系统设置":
    st.title("系统设置")
    
    st.write("### 系统信息")
    st.write(f"当前时间: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
    st.write(f"后端地址: http://localhost:8765")
    st.write(f"大屏地址: http://localhost:8501")
    
    if st.button("刷新数据"):
        st.cache_data.clear()
        st.rerun()

# 底部信息
st.divider()
st.caption("Project Claw v1.0 | 融资路演大屏 | 🦞")
```
